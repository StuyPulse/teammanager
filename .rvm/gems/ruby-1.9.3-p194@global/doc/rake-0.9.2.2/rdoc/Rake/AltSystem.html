def find_runnable(file)
  if file =~ RUNNABLE_PATTERN
    file
  else
    RUNNABLE_EXTS.each { |ext|
      if File.exist?(test = "#{file}.#{ext}")
        return test
      end
    }
    nil
  end
end

def repair_command(cmd)
  "call " + (
    if cmd =~ %\A\s*\".*?\"!
      # already quoted
      cmd
    elsif match = cmd.match(%\A\s*(\S+)!)
      if match[1] =~ %/!
        # avoid x/y.bat interpretation as x with option /y
        %"#{match[1]}"! + match.post_match
      else
        # a shell command will fail if quoted
        cmd
      end
    else
      # empty or whitespace
      cmd
    end
  )
end

def system(cmd, *args)
  repaired = (
    if args.empty?
      [repair_command(cmd)]
    elsif runnable = find_runnable(cmd)
      [File.expand_path(runnable), *args]
    else
      # non-existent file
      [cmd, *args]
    end
  )
  kernel_system(*repaired)
end
