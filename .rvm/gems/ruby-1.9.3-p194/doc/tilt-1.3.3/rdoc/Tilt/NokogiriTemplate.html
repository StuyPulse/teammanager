def evaluate(scope, locals, &block)
  block &&= proc { yield.gsub(%r^<\?xml version=\"1\.0\"\?>\n?/, "") }

  if data.respond_to?(:to_str)
    super(scope, locals, &block)
  else
    ::Nokogiri::XML::Builder.new.tap(&data).to_xml
  end
end
