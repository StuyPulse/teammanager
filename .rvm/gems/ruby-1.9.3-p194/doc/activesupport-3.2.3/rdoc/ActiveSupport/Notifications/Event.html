def initialize(name, start, ending, transaction_id, payload)
  @name           = name
  @payload        = payload.dup
  @time           = start
  @transaction_id = transaction_id
  @end            = ending
  @duration       = 1000.0 * (@end - @time)
end

def parent_of?(event)
  start = (time - event.time) * 1000
  start <= 0 && (start + duration >= event.duration)
end
