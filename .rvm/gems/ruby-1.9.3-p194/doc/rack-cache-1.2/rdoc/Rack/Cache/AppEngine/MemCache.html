def put(key, value, ttl = nil)
  expiration = ttl ? MC::Expiration.byDeltaSeconds(ttl) : nil
  value = Base64.encode64(Marshal.dump(value)).gsub(%r\n/, '')
  MC::Service.put(key, value, expiration)
end
