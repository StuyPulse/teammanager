def compile_update values
  um = UpdateManager.new @engine

  if Nodes::SqlLiteral === values
    relation = @ctx.from
  else
    relation = values.first.first.relation
  end
  um.table relation
  um.set values
  um.take @ast.limit.expr if @ast.limit
  um.order(*@ast.orders)
  um.wheres = @ctx.wheres
  um
end
